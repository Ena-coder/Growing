搞清楚如何决定那个字符高亮？
生成26个字母的方法：

```js
var indexList = function () {
  var indexList = []; //存储字母的数组容器
  var charCodeOfA = 'A'.charCodeAt(0); //charCodeAt(0)返回第0个字符的Unicode 编码   =>65
  for (var i = 0; i < 26; i++) {
    indexList.push(String.fromCharCode(charCodeOfA + i)); //String.fromCharCode将Unicode编码转为一个字符串
  }
  return indexList;
};
```
